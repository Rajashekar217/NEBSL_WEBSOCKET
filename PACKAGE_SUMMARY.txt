================================================================================
COMMON TASKS
================================================================================

Task: Subscribe to tokens
────────────────────────────────────────────────────────────────────────────

  from nse_ws_client import NSEWebSocketClient
  
  client = NSEWebSocketClient()
  
  # Equity (CM)
  client.subscribe_cm({26000: "NIFTY50"})
  
  # Derivatives (FO)
  client.subscribe_fo({75663: "BANKNIFTY_FUT"})
  
  client.start()

Task: Get Latest Price
────────────────────────────────────────────────────────────────────────────

  q = client.data_channel()
  data = q.get()
  print(f"LTP: {data.ltp}")

Task: Analyze Order Book
────────────────────────────────────────────────────────────────────────────

  for level in data.bid_levels:
      print(f"Bid: {level['price']} x {level['qty']}")
  
  for level in data.ask_levels:
      print(f"Ask: {level['price']} x {level['qty']}")

Task: Monitor Connection Health
────────────────────────────────────────────────────────────────────────────

  status = client.get_status()
  if not status['connected']:
      print("Connection lost")
  
  if status['data_flow_seconds'] > 5:
      print("No recent data")

Task: Calculate Basis (Spot vs Futures)
────────────────────────────────────────────────────────────────────────────

  latest = {}
  while True:
      data = q.get()
      latest[data.token] = data
      
      if 26000 in latest and 75697 in latest:
          basis = latest[75697].ltp - latest[26000].ltp
          print(f"Basis: {basis}")
